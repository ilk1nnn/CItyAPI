        function Login() {
            let form = document.forms['myform'];
            let username = form['username'].value;
            let password = form['username'].value;
            SendLoginRequest(username, password);



        }
        let token = "";
        function SendLoginRequest(username, password) {
            let obj = {
                username: username,
                password: password
            };
            $.ajax({
                type: "POST",
                url: "https://localhost:7100/api/Auth/login",   ///// URL must be specified 
                contentType: "application/json; charset=utf-8",
                dataType: "json",
                data: JSON.stringify(obj), //this class returns a json object
                cache: false,
                success: function (result) {
                    token = result;
                    console.log(token);
                    alert("Logined Successfully!");
                    let div2 = document.getElementById("div2");
                    let div3 = document.getElementById("div3");
                    let div4 = document.getElementById("div4");
                    let div5 = document.getElementById("div5");
                    div2.style.display = "block";
                    div3.style.display = "block";
                    div4.style.display = "block";
                    div5.style.display = "block";
                },
                error: function (ex) {
                    alert("Username Or Password Is Invalid!");
                }
            });
        }


        function GetCity() {
            SendCityRequest();
        }
        let cities = "";
        function SendCityRequest() {
            $.ajax({
                type: "GET",
                url: "https://localhost:7100/api/Cities",
                contentType: "application/json; charset=utf-8",
                dataType: "json",
                cache: false,
                headers: { "Authorization": 'Bearer ' + token },
                success: function (result) {
                    cities = result;
                    if (result == null) {
                        alert("You must login For Getting Elements");
                    }
                    else {
                        alert(JSON.stringify(result));
                        console.log(JSON.stringify(result));
                    }
                },
                error: function (ex) {
                    alert(ex.responseText);
                }
            });
        }


        function AddCity() {
            let form = document.forms['myform3'];
            let Description = form['Description'].value;
            let Name = form['Name'].value;
            let UserId = form['UserId'].value;
            SendAddCityRequest(Description, Name, UserId);
        }

        function SendAddCityRequest(description, namecity, userid) {
            
            var newData = {};
            newData.Description = description;
            newData.Name = namecity;
            newData.UserId = userid;

            var jsonData = JSON.stringify(newData);

            $.ajax({
                    type: "POST",
                    url: "https://localhost:7100/api/Cities/add",
                    data: jsonData,
                    contentType: "application/json; charset=utf-8",
                    dataType: "json",
                    headers: { "Authorization": 'Bearer ' + token },
                    success: function (result) {
                        alert("Added Successfully : " + JSON.stringify(result));
                        console.log(JSON.stringify(result));
                    },
                    error: function (ex) {
                        alert(ex.responseText);
                    }
                });

            
        }
